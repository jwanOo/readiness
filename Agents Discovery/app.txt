# --- Web-Scraping Funktion für SAP-Detailseiten ---
def lese_sap_detail_seite(url: str) -> str:
    import subprocess, sys, os
    clean_url = url.strip()
    if not clean_url.startswith("http"): clean_url = "https://" + clean_url
    script_code = f"""
import sys
try:
    from playwright.sync_api import sync_playwright
    with sync_playwright() as p:
        browser = p.chromium.launch(headless=True, args=["--disable-blink-features=AutomationControlled"])
        context = browser.new_context(user_agent="Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36")
        page = context.new_page()
        page.goto("{clean_url}", wait_until="load", timeout=45000)
        page.wait_for_timeout(5000)
        page.evaluate('document.querySelectorAll(\\'[id*="trustarc"], [class*="overlay"], .truste-banner\\').forEach(el => el.remove()); document.body.style.overflow = \\'auto\\';')
        content = page.locator("body").inner_text()
        print(content[:15000])
        browser.close()
except Exception as e:
    print("ERROR_SCRAPER:", str(e))
"""
    script_path = "temp_scraper.py"
    with open(script_path, "w", encoding="utf-8") as f: f.write(script_code)
    try:
        result = subprocess.run([sys.executable, script_path], capture_output=True, text=True, encoding="utf-8", timeout=60)
        if os.path.exists(script_path): os.remove(script_path)
        if "ERROR_SCRAPER:" in result.stdout: return f"Fehler auf der Seite: {result.stdout}"
        if not result.stdout.strip(): return "Seite geladen, aber Inhalt leer."
        return result.stdout.strip()
    except Exception as e:
        if os.path.exists(script_path): os.remove(script_path)
        return f"Systemfehler beim isolierten Aufruf: {str(e)}"

# --- Synchronisationslogik für den Datenabgleich ---
def perform_data_merge():
    try:
        subprocess.run([sys.executable, "downloader.py"], check=True)
        if os.path.exists(RAW_FILE):
            df_raw = pd.read_csv(RAW_FILE, sep=';').fillna('')
            df_raw.columns = df_raw.columns.str.strip()
            
            # Bestimmung der Identifier-Spalte
            if "Identifier" in df_raw.columns:
                id_col = "Identifier"
            else:
                id_col = df_raw.columns[0] # Fallback auf erste Spalte
                
            sap_columns = [col for col in df_raw.columns if col != id_col]
            
            if os.path.exists(DB_FILE):
                df_db = pd.read_csv(DB_FILE).fillna('')
                # Anpassung alter Spaltennamen (Migration)
                if "Neu hinzugefügt" in df_db.columns:
                    df_db = df_db.rename(columns={"Neu hinzugefügt": STATUS_COLUMN})
                    df_db[STATUS_COLUMN] = df_db[STATUS_COLUMN].replace({"🌟 NEU": "🌟 NEW", "🔄 AKTUALISIERT": "🔄 UPDATED", "🌟 NEU (Initial)": "🌟 NEW (Initial)"})
                if "Änderungshistorie (Auto)" in df_db.columns:
                    df_db = df_db.rename(columns={"Änderungshistorie (Auto)": HISTORY_COLUMN})
                if HISTORY_COLUMN not in df_db.columns: df_db[HISTORY_COLUMN] = ""
                if "Tags" not in df_db.columns: df_db["Tags"] = ""
                
                # Abgleich der Identifier bei Migration
                if id_col not in df_db.columns and id_col == "Identifier":
                    pass 
                
                for index, raw_row in df_raw.iterrows():
                    use_case_id = raw_row[id_col]
                    
                    if id_col in df_db.columns:
                        db_mask = df_db[id_col] == use_case_id
                    else:
                        db_mask = pd.Series([False]*len(df_db))
                    
                    if db_mask.any():
                        db_idx = df_db.index[db_mask].tolist()[0]
                        is_updated = False
                        changes_list = []
                        
                        for col in sap_columns:
                            if col in df_db.columns:
                                val_raw = str(raw_row[col]).replace('nan', '').strip()
                                val_db = str(df_db.at[db_idx, col]).replace('nan', '').strip()
                                
                                if val_raw != val_db:
                                    old_str = val_db if len(val_db) <= 40 else val_db[:37] + "..."
                                    new_str = val_raw if len(val_raw) <= 40 else val_raw[:37] + "..."
                                    if not old_str: old_str = "Leer"
                                    if not new_str: new_str = "Leer"
                                    changes_list.append(f"'{col}': '{old_str}' -> '{new_str}'")
                                    df_db.at[db_idx, col] = raw_row[col]
                                    is_updated = True
                        
                        if is_updated:
                            if df_db.at[db_idx, STATUS_COLUMN] not in ["🌟 NEW", "🌟 NEW (Initial)"]:
                                df_db.at[db_idx, STATUS_COLUMN] = "🔄 UPDATED"
                            date_str = datetime.now().strftime("%d.%m.%Y %H:%M")
                            new_hist_entry = f"[{date_str}] " + " | ".join(changes_list)
                            old_hist = str(df_db.at[db_idx, HISTORY_COLUMN]).replace('nan', '').strip()
                            df_db.at[db_idx, HISTORY_COLUMN] = (new_hist_entry + " || " + old_hist) if old_hist else new_hist_entry
                    else:
                        new_row_dict = raw_row.to_dict()
                        for col in EDITABLE_COLUMNS: new_row_dict[col] = ""
                        new_row_dict[STATUS_COLUMN] = "🌟 NEW"
                        new_row_dict[HISTORY_COLUMN] = f"[{datetime.now().strftime('%d.%m.%Y')}] Newly added"
                        df_db = pd.concat([df_db, pd.DataFrame([new_row_dict])], ignore_index=True)
                        
                if "Business Value according to SAP" in df_db.columns:
                    df_db = df_db.drop(columns=["Business Value according to SAP"])
                df_db.to_csv(DB_FILE, index=False)
            else:
                # Initialisierung der Datenbank falls Datei nicht existiert
                for col in EDITABLE_COLUMNS: df_raw[col] = ""
                df_raw[STATUS_COLUMN] = "🌟 NEW (Initial)"
                df_raw[HISTORY_COLUMN] = ""
                if "Business Value according to SAP" in df_raw.columns:
                    df_raw = df_raw.drop(columns=["Business Value according to SAP"])
                df_raw.to_csv(DB_FILE, index=False)
                
        current_config = load_config()
        current_config["last_sync_time"] = datetime.now().strftime("%d.%m.%Y %H:%M:%S")
        save_config(current_config)
        st.session_state.gemini_chat = None 
        return True
    except Exception as e:
        print(f"Sync Fehler: {e}")
        return False

# --- Ausführung des Hintergrund-Tasks für den Scheduler ---
def background_task():
    perform_data_merge()

# --- Management des Hintergrund-Threads für Zeitsteuerung ---
@st.cache_resource
def get_scheduler_manager():
    class SchedulerManager:
        def __init__(self):
            self.thread = None
            self.stop_event = threading.Event()
        def start(self, time_str):
            self.stop()
            self.stop_event.clear()
            self.thread = threading.Thread(target=self._loop, args=(time_str,), daemon=True)
            self.thread.start()
        def stop(self):
            self.stop_event.set()
            if self.thread: self.thread.join(timeout=1)
        def _loop(self, target_time):
            schedule.clear()
            schedule.every().day.at(target_time).do(background_task)
            while not self.stop_event.is_set():
                schedule.run_pending()
                time_lib.sleep(10)
    return SchedulerManager()

scheduler = get_scheduler_manager()
